```js
const ShaderLib = {
  // 从 activetheory.net/assets/shaders/compiled.vs 直接搬运
  curlNoiseGLSL: `
    vec3 snoiseVec3(vec3 x) {
      float s = sin(vec2(x.x*1.8+x.z*3., x.y*1.8+x.x*3.)) +
                sin(vec2(x.z*1.8+x.y*3., x.x*4.8+x.z*4.5)) +
                sin(vec2(x.y*(-7.0)+x.z*1.2, x.z*(-5.0)+x.y*2.13));
      return vec3(s, sin(x.y*(-0.48)+x.z*5.4)+sin(x.z*2.56+x.y*5.4), sin(x.z*4.16+x.x*2.4)+sin(x.x*(-4.16)+x.z*1.35));
    }

    vec3 curlNoise(vec3 p) {
      float e = 0.01;
      vec3 dx = vec3(e, 0.0, 0.0);
      vec3 dy = vec3(0.0, e, 0.0);
      vec3 dz = vec3(0.0, 0.0, e);
      vec3 p_x0 = snoiseVec3(p - dx);
      vec3 p_x1 = snoiseVec3(p + dx);
      vec3 p_y0 = snoiseVec3(p - dy);
      vec3 p_y1 = snoiseVec3(p + dy);
      vec3 p_z0 = snoiseVec3(p - dz);
      vec3 p_z1 = snoiseVec3(p + dz);
      float x = p_y1.z - p_y0.z - p_z1.y + p_z0.y;
      float y = p_z1.x - p_z0.x - p_x1.z + p_x0.z;
      float z = p_x1.y - p_x0.y - p_y1.x + p_y0.x;
      return normalize(vec3(x, y, z)) / (2.0 * e);
    }
  `,

  // RGB 色散（直接抄 rgbshift.fs）
  rgbShiftFS: `
    uniform sampler2D tDiffuse;
    uniform float uRGBAngle;
    uniform float uRGBAmount;
    varying vec2 vUv;

    void main() {
      vec2 offset = vec2(cos(uRGBAngle), sin(uRGBAngle)) * uRGBAmount;
      float r = texture2D(tDiffuse, vUv + offset).r;
      float g = texture2D(tDiffuse, vUv).g;
      float b = texture2D(tDiffuse, vUv - offset).b;
      float a = texture2D(tDiffuse, vUv).a;
      gl_FragColor = vec4(r, g, b, a);
    }
  `,

  // HSV 转换（直接抄 rgb2hsv.fs）
  hsvGLSL: `
    vec3 rgb2hsv(vec3 c) {
      vec4 K = vec4(0.0, -1.0/3.0, 2.0/3.0, -1.0);
      vec4 p = mix(vec4(c.bg, K.wz), vec4(c.gb, K.xy), step(c.b, c.g));
      vec4 q = mix(vec4(p.xyw, c.r), vec4(c.r, p.yzx), step(p.x, c.r));
      float d = q.x - min(q.w, q.y);
      return vec3(abs(q.z + (q.w - q.y)/(6.0*d + 1e-10)), d/(q.x + 1e-10), q.x);
    }
    vec3 hsv2rgb(vec3 c) {
      vec4 K = vec4(1.0, 2.0/3.0, 1.0/3.0, 3.0);
      vec3 p = abs(fract(c.xxx + K.xyz) * 6.0 - K.www);
      return c.z * mix(K.xxx, clamp(p - K.xxx, 0.0, 1.0), c.y);
    }
  `,

  // 混合模式（直接抄 blendmodes.glsl）
  blendModesGLSL: `
    float blendColorDodge(float base, float blend) {
      return (blend == 1.0) ? blend : min(base/(1.0-blend), 1.0);
    }
    float blendGlow(float base, float blend) {

    uniform float uTime;
    uniform float uPointScale;
    void main() {

      vec4 mvPosition = modelViewMatrix * vec4(position, 1.0);
      gl_PointSize = aSize * uPointScale * (300.0 / -mvPosition.z);
      gl_Position = projectionMatrix * mvPosition;
    }
  `,
```

```js
    // Scene
    this.scene = new THREE.Scene();
    this.scene.background = new THREE.Color(COLORS.bg);

    // Camera
    this.camera = new THREE.PerspectiveCamera(45, this._width / this._height, 0.1, 100);
    this.camera.position.set(0, 0, 20);

    // Clock
    this.clock = new THREE.Clock();

    // 子模块占位
    this.modules = {};

    // 事件总线
    this._events = {};

    // 绑定 resize
    this._onResize = this._onResize.bind(this);
    window.addEventListener('resize', this._onResize);
  }

  on(event, fn) {
    (this._events[event] ||= []).push(fn);
  }

  emit(event, data) {
    (this._events[event] || []).forEach(fn => fn(data));
  }

  _onResize() {
    this._width = window.innerWidth;
    this._height = window.innerHeight;
    this.renderer.setSize(this._width, this._height);
    this.camera.aspect = this._width / this._height;
    this.camera.updateProjectionMatrix();
    this.emit('resize', { width: this._width, height: this._height });
  }

  get currentSceneIndex() { return this._sceneIndex; }
}
```

```js
class MouseManager {
  constructor(canvas) {
    this._canvas = canvas;
    this.normalized = { x: 0, y: 0 };
    this.screen = { x: 0, y: 0 };
    this.velocity = { x: 0, y: 0 };
    this.ripple = { x: 0, y: 0, strength: 0 };
    this._prevX = 0;
    this._prevY = 0;

    this._onMove = (e) => {
      this.screen.x = e.clientX;
      this.screen.y = e.clientY;
      this.normalized.x = (e.clientX / window.innerWidth) * 2 - 1;
      this.normalized.y = -(e.clientY / window.innerHeight) * 2 + 1;
      this.ripple.x = this.normalized.x;
      this.ripple.y = this.normalized.y;
    };

    this._onLeave = () => {
      this.ripple.strength = 0;
    };

    window.addEventListener('mousemove', this._onMove, { passive: true });
    window.addEventListener('mouseleave', this._onLeave);
  }

  update() {
    this.velocity.x = this.normalized.x - this._prevX;
    this.velocity.y = this.normalized.y - this._prevY;
    this._prevX = this.normalized.x;
    this._prevY = this.normalized.y;

    const speed = Math.sqrt(this.velocity.x ** 2 + this.velocity.y ** 2);
    this.ripple.strength = Math.min(speed * 3, 1.0);
  }

  dispose() {
    window.removeEventListener('mousemove', this._onMove);
    window.removeEventListener('mouseleave', this._onLeave);
  }
}
```

```js
class CameraController {
  constructor(camera) {
    this._camera = camera;
    this._keyframes = [];  // { progress, position, lookAt, fov }
    this._currentPos = new THREE.Vector3();
    this._currentLookAt = new THREE.Vector3();
    this._currentFov = 45;
  }

  addKeyframe(progress, position, lookAt, fov) {
    this._keyframes.push({
      progress: THREE.MathUtils.clamp(progress, 0, 1),
      position: position.clone(),
      lookAt: lookAt.clone(),
      fov: fov || 45
    });
    this._keyframes.sort((a, b) => a.progress - b.progress);
  }

  update(scrollProgress) {
    const t = THREE.MathUtils.clamp(scrollProgress, 0, 1);

    // 找到前后最近的两个关键帧
    let prev = this._keyframes[0];
    let next = this._keyframes[this._keyframes.length - 1];

    for (let i = 0; i < this._keyframes.length - 1; i++) {
      if (t >= this._keyframes[i].progress && t <= this._keyframes[i + 1].progress) {
        prev = this._keyframes[i];
        next = this._keyframes[i + 1];
        break;
      }
    }

    const range = next.progress - prev.progress;
    const localT = range > 0 ? (t - prev.progress) / range : 0;
    const ease = localT < 0.5
      ? 2 * localT * localT
      : 1 - Math.pow(-2 * localT + 2, 2) / 2;  // easeInOutQuad

    this._currentPos.lerpVectors(prev.position, next.position, ease);
    this._currentLookAt.lerpVectors(prev.lookAt, next.lookAt, ease);
    this._currentFov = THREE.MathUtils.lerp(prev.fov, next.fov, ease);

    this._camera.position.copy(this._currentPos);
    this._camera.lookAt(this._currentLookAt);
    this._camera.fov = this._currentFov;
    this._camera.updateProjectionMatrix();
  }

  getPosition() { return this._camera.position.clone(); }
  getTarget() { return this._currentLookAt.clone(); }
}
```

```js
class StarfieldBackground {
  constructor(scene, count = 1_200_000) {
    this._scene = scene;
    this._count = count;

    const positions = new Float32Array(count * 3);
    const colors = new Float32Array(count * 3);
    const sizes = new Float32Array(count);

    for (let i = 0; i < count; i++) {
      // 球形分布，半径 30-60
      const r = 30 + Math.random() * 30;
      const theta = Math.random() * Math.PI * 2;
      const phi = Math.acos(2 * Math.random() - 1);
      positions[i * 3] = r * Math.sin(phi) * Math.cos(theta);
      positions[i * 3 + 1] = r * Math.sin(phi) * Math.sin(theta);
      positions[i * 3 + 2] = r * Math.cos(phi);

      // 冷蓝白微光
      const brightness = 0.3 + Math.random() * 0.4;
      colors[i * 3] = 0.5 * brightness;
      colors[i * 3 + 1] = 0.55 * brightness;
      colors[i * 3 + 2] = 0.65 * brightness;

      sizes[i] = 0.02 + Math.random() * 0.06;
    }

    const geometry = new THREE.BufferGeometry();
    geometry.setAttribute('position', new THREE.BufferAttribute(positions, 3));
    geometry.setAttribute('color', new THREE.BufferAttribute(colors, 3));
    geometry.setAttribute('aSize', new THREE.BufferAttribute(sizes, 1));

    const material = new THREE.ShaderMaterial({
      uniforms: {
        uTime: new THREE.Uniform(0),
        uPointScale: new THREE.Uniform(1.0),
      },
      vertexShader: ShaderLib.particleVS,
      fragmentShader: ShaderLib.particleFS,
      transparent: true,
      depthWrite: false,
      blending: THREE.AdditiveBlending,
    });

    this._points = new THREE.Points(geometry, material);
    this._points.name = 'starfield';
    scene.add(this._points);
  }

  update(delta, cameraPos = null, mouseRipple = null) {
    this._points.material.uniforms.uTime.value += delta;
  }

  dispose() {
    this._points.geometry.dispose();
    this._points.material.dispose();
    this._scene.remove(this._points);
  }
}
```

```js
class SynapseNetwork {
  constructor(scene) {

  addNode(id, position, options = {}) {

    mesh.name = `node_${id}`;

    // 光晕（外层透明球）
    const glowGeo = new THREE.SphereGeometry(radius * 2.5, 16, 16);
    const glowMat = new THREE.ShaderMaterial({
      uniforms: { uColor: new THREE.Uniform(new THREE.Color(colorHex)) },
      vertexShader: `
        varying vec3 vNormal; varying vec3 vPosition;
        void main() {
          vNormal = normalize(normalMatrix * normal);
          vPosition = position;
          gl_Position = projectionMatrix * modelViewMatrix * vec4(position, 1.0);
        }
      `,
      fragmentShader: ShaderLib.glowFS,
      transparent: true,
      depthWrite: false,
      blending: THREE.AdditiveBlending,
    });
    const glowMesh = new THREE.Mesh(glowGeo, glowMat);
    glowMesh.name = `glow_${id}`;

    mesh.add(glowMesh);
    this._group.add(mesh);

    this._nodes.set(id, { id, position: position.clone(), mesh, glow: glowMesh, isCore, radius, _breathPhase: Math.random() * Math.PI * 2 });
    return id;
  }

  connect(fromId, toId, options = {}) {
    const from = this._nodes.get(fromId);
    const to = this._nodes.get(toId);
    if (!from || !to) return;

    const { tubeRadius = 0.02, flowSpeed = 1.0 } = options;

    // 贝塞尔曲线：起点 → 中点抬高 → 终点
    const mid = new THREE.Vector3().addVectors(from.position, to.position).multiplyScalar(0.5);
    mid.y += from.position.distanceTo(to.position) * 0.3;

    const curve = new THREE.QuadraticBezierCurve3(from.position.clone(), mid, to.position.clone());
    const tubeGeo = new THREE.TubeGeometry(curve, 32, tubeRadius, 6, false);
    const tubeMat = new THREE.MeshBasicMaterial({
      color: COLORS.darkGold,
      transparent: true,
      opacity: 0.15,
      depthWrite: false,
    });
    const tube = new THREE.Mesh(tubeGeo, tubeMat);
    tube.name = `tube_${fromId}_${toId}`;
    this._group.add(tube);

    // 管道内流动粒子（沿曲线采样）
    const flowCount = 60;
    const flowPositions = new Float32Array(flowCount * 3);
    const flowColors = new Float32Array(flowCount * 3);
    const flowSizes = new Float32Array(flowCount);

    for (let i = 0; i < flowCount; i++) {
      const t = i / flowCount;
      const pt = curve.getPointAt(t);
      flowPositions[i * 3] = pt.x;
      flowPositions[i * 3 + 1] = pt.y;
      flowPositions[i * 3 + 2] = pt.z;
      flowColors[i * 3] = 0.83;
      flowColors[i * 3 + 1] = 0.66;
      flowColors[i * 3 + 2] = 0.26;
      flowSizes[i] = 0.015;
    }

    const flowGeo = new THREE.BufferGeometry();
    flowGeo.setAttribute('position', new THREE.BufferAttribute(flowPositions, 3));
    flowGeo.setAttribute('aColor', new THREE.BufferAttribute(flowColors, 3));
    flowGeo.setAttribute('aSize', new THREE.BufferAttribute(flowSizes, 1));

    const flowMat = new THREE.ShaderMaterial({
      uniforms: { uTime: new THREE.Uniform(0), uPointScale: new THREE.Uniform(1.5) },
      vertexShader: ShaderLib.particleVS,
      fragmentShader: ShaderLib.particleFS,
      transparent: true,
      depthWrite: false,
      blending: THREE.AdditiveBlending,
    });

    const flowParticles = new THREE.Points(flowGeo, flowMat);
    flowParticles.name = `flow_${fromId}_${toId}`;
    this._group.add(flowParticles);

    this._edges.push({ from: fromId, to: toId, tube, particles: flowParticles, curve, flowSpeed, _pulsePhase: Math.random() * Math.PI * 2 });
  }

  setNodeActive(id, active) {
    const node = this._nodes.get(id);
    if (!node) return;
    node.glow.material.uniforms.uColor.value.set(active ? COLORS.mainGold : COLORS.amber);
    node.mesh.material.color.set(active ? COLORS.mainGold : COLORS.amber);
  }

  getNodeWorldPosition(id) {
    const node = this._nodes.get(id);
    if (!node) return null;
    const pos = new THREE.Vector3();
    node.mesh.getWorldPosition(pos);
    return pos;
  }

  update(delta, mousePos = null) {
    const t = performance.now() * 0.001;

    // 节点呼吸
    this._nodes.forEach((node, id) => {
      const breath = 1 + Math.sin(t * 2.0 + node._breathPhase) * 0.05;
      node.mesh.scale.setScalar(breath);
      node.glow.scale.setScalar(1 / breath);
    });

    // 管道粒子流动
    this._edges.forEach(edge => {
      const positions = edge.particles.geometry.attributes.position.array;
      for (let i = 0; i < positions.length / 3; i++) {
        let tVal = (i / (positions.length / 3) + t * edge.flowSpeed * 0.1) % 1;
        const pt = edge.curve.getPointAt(tVal);
        positions[i * 3] = pt.x;
        positions[i * 3 + 1] = pt.y;
        positions[i * 3 + 2] = pt.z;
      }
      edge.particles.geometry.attributes.position.needsUpdate = true;
      edge.particles.material.uniforms.uTime.value = t;
    });
  }

  clear() {
    while (this._group.children.length > 0) {
      const child = this._group.children[0];
      if (child.geometry) child.geometry.dispose();
      if (child.material) {
        if (Array.isArray(child.material)) child.material.forEach(m => m.dispose());
        else child.material.dispose();
      }
      this._group.remove(child);
    }
    this._nodes.clear();
    this._edges = [];
  }

  dispose() {
    this.clear();
    this._scene.remove(this._group);
  }
}
```

```js
      new THREE.Vector3(0, 2, 12), new THREE.Vector3(0, 0, 0), 42);
    this.cameraController.addKeyframe(0.75,
      new THREE.Vector3(0, 1.5, 3), new THREE.Vector3(0, 0, 0), 35);
    // 场景四：关于（拉远）
    this.cameraController.addKeyframe(0.9,
      new THREE.Vector3(0, 0, 18), new THREE.Vector3(0, 0, 0), 50);
    this.cameraController.addKeyframe(1.0,
      new THREE.Vector3(0, 0, 15), new THREE.Vector3(0, 0, 0), 45);

    // 3. 场景初始化
    this.scenes = [
      new Scene1_Hero(this),
      new Scene2_NeuralGallery(this),
      new Scene3_SkillConstellation(this),
      new Scene4_About(this),
    ];
    this.scenes[0].init();
    this.scenes[1].init(DataModel.projects);
    this.scenes[2].init(DataModel.skills);
    this.scenes[3].init(DataModel.personal);

    // 4. 设置 ScrollTrigger
    this._setupScroll();

    // 5. 开始渲染循环
    this._animate();
  }

  _setupScroll() {
    // 使用 GSAP ScrollTrigger 驱动
    // 创建一个代理滚动容器
    const scrollProxy = { progress: 0 };

    gsap.registerPlugin(ScrollTrigger);
    gsap.to(scrollProxy, {
      progress: 1,
      ease: 'none',
      scrollTrigger: {
        trigger: document.body,
        start: 'top top',
        end: 'bottom bottom',
        scrub: 0.5,
        onUpdate: (self) => {
          this.cameraController.update(self.progress);
          this._updateScenes(self.progress);
        }
      }
    });

    // 设置 body 足够长以支持滚动
    document.body.style.height = '400vh';
  }

  _updateScenes(progress) {
    const perScene = 1 / 4;

    for (let i = 0; i < this.scenes.length; i++) {
      const sceneStart = i * perScene;
      const sceneEnd = (i + 1) * perScene;

      if (progress >= sceneStart && progress <= sceneEnd) {
        const localProgress = (progress - sceneStart) / perScene;
        this.scenes[i].update(this.clock.getDelta(), localProgress, this.mouseManager.normalized);
      }
    }
  }

  _animate() {
    requestAnimationFrame(() => this._animate());

    const delta = this.clock.getDelta();
    this.mouseManager.update();
    this.cursorUI.update(this.mouseManager);
    this.starfield.update(delta, this.cameraController.getPosition(), this.mouseManager.ripple);

    this.renderer.render(this.scene, this.camera);
  }

  destroy() {
    this.scenes.forEach(s => s.dispose());
    this.starfield.dispose();
    this.mouseManager.dispose();
    this.cursorUI.dispose();
    this.renderer.dispose();
  }
}
```

```js
function generateMatcapTexture() {
  const size = 256;
  const canvas = document.createElement('canvas');
  canvas.width = size;
  canvas.height = size;
  const ctx = canvas.getContext('2d');
  // 径向渐变模拟球形光照
  const gradient = ctx.createRadialGradient(size*0.3, size*0.3, size*0.1, size*0.5, size*0.5, size*0.7);
  gradient.addColorStop(0, '#FFF8E7');
  gradient.addColorStop(0.3, '#D4A843');
  gradient.addColorStop(0.7, '#4A3000');
  gradient.addColorStop(1, '#000000');
  ctx.fillStyle = gradient;
  ctx.fillRect(0, 0, size, size);
  return new THREE.CanvasTexture(canvas);
}
```
